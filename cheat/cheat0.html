Some Helpful kubectl commands

https://kubernetes.io/docs/reference/kubectl/cheatsheet/

//show all deployments, services, pods and replicasets in a namespace
$ kubectl version --short
$ kubectl cluster-info $ kubectl cluster-info | grep -i dns $ kubectl get all -n
$ kubectl config view
$ kubectl config get-contexts
$ kubectl config set-context tomwat-cluster --user=cluster-admin --namespace=apic \
&& kubectl config use-context tomwat-cluster
$ kubectl exec busybox nslookup kubernetes.default
